// Example: Find unique string in array
function findUniq(arr) {
  const normalize = str => [...new Set(str.toLowerCase().replace(/\s+/g, '').split(''))].sort().join('');
  const signatures = arr.map(normalize);
  for(let i=0;i<signatures.length;i++) {
    if(signatures.indexOf(signatures[i]) === signatures.lastIndexOf(signatures[i])) return arr[i];
  }
}
